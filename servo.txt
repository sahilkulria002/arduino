#include <ESP32Servo.h>

Servo myservo;
int servoPin = 18;

void setup() {
  Serial.begin(115200);
  myservo.attach(servoPin);
  myservo.write(0);
  Serial.println("Servo initialized at 0Â°");
}

void loop() {
  for (int pos = 0; pos <= 180; pos += 10) {
    myservo.write(pos);
    Serial.println(pos);
    delay(200);
  }
  for (int pos = 180; pos >= 0; pos -= 10) {
    myservo.write(pos);
    Serial.println(pos);
    delay(200);
  }
}
